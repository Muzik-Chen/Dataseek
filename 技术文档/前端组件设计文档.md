## 1. 前端技术概览

| 层级 | 技术 | 版本 | 用途 |
|------|------|------|------|
| 框架 | Vue 3 (Composition API) | 3.5.x | 全部页面与组件 |

| UI 库 | Element Plus | 2.9.x | 按钮/表单/表格/弹窗/布局/图标 |

| HTTP | Axios | 1.7.x | REST API 调用 + SSE 流式接收 |
| 路由 | Vue Router | 4.4.x | 页面路由 + 导航守卫 |
| 语言 | JavaScript | ES2022+ | `<script setup>` 语法 |
| 样式 | CSS + Element Plus 主题变量 | — | 全局样式 + 组件样式 |

router.beforeEach((to, from, next) => {
  const userStore = useUserStore()

  // 公开浏览页面（列表、详情、Feed流、日历、大屏、客服、搜索）
  const publicPrefixes = [
    '/', '/foods', '/heritages', '/festival', '/dashboard',
    '/login', '/register', '/chat', '/community', '/search'
  ]
  const isPublicPage = publicPrefixes.some(p => to.path.startsWith(p))

  // 需要登录
  if (to.meta.requiresAuth && !userStore.isLoggedIn) {
    return next('/login?redirect=' + encodeURIComponent(to.fullPath))
  }

  // 游客页面（已登录用户访问登录/注册 → 跳首页）
  if (to.meta.guest && userStore.isLoggedIn) {
    return next('/')
  }

  // 需要管理员
  if (to.meta.requiresAdmin && userStore.user?.role !== 'admin') {
    return next('/')
  }

  next()
})

export default router

| 降级方案 | LLM 失败时使用动态模板（从 DB 读取真实数据拼行程），标记 "⚠️ 离线演示版" |

|---------|------|

## 8. 状态管理设计

### 8.1 userStore

```javascript
// stores/user.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { authApi, userApi } from '@/api'

export const useUserStore = defineStore('user', () => {
  // ===== State =====
  const token = ref(localStorage.getItem('token') || '')
  const user = ref(null)

  // ===== Getters =====
  const isLoggedIn = computed(() => !!token.value)
  const isAdmin = computed(() => user.value?.role === 'admin')

  // ===== Actions =====
  async function login(email, password) {
    const res = await authApi.login(email, password)
    token.value = res.token
    user.value = res.user
    localStorage.setItem('token', res.token)
  }

  async function register(email, code, password, confirmPassword, nickname) {
    const res = await authApi.register(email, code, password, confirmPassword, nickname)
    token.value = res.token
    user.value = res.user
    localStorage.setItem('token', res.token)
  }

  function logout() {
    token.value = ''
    user.value = null
    localStorage.removeItem('token')
    router.push('/')

  let saveTimer = null
  watch(
    [form, streamPlans, streamNarration, activeResultPlanId],
    () => {
      if (saveTimer) clearTimeout(saveTimer)
      saveTimer = setTimeout(saveToStorage, 300)
    },
    { deep: true }

  function clearAll() {

    streamPlans.value = []
    streamNarration.value = ''
    activeResultPlanId.value = null
    hasSavedPlans.value = false
    try { localStorage.removeItem(STORAGE_KEY) } catch { /* ignore */ }

  return { form, streamPlans, streamNarration, activeResultPlanId, hasSavedPlans, clearAll }

// api/request.js
import axios from 'axios'
import { useUserStore } from '@/stores/user'
import { toast } from '@/utils/toast'
import router from '@/router'

const api = axios.create({
  baseURL: import.meta.env.VITE_API_BASE || '/api/v1',
  timeout: 30000,  // 普通接口 30s 超时（SSE 除外）
})

// 请求拦截器：自动注入 Token
api.interceptors.request.use(config => {
  const userStore = useUserStore()
  if (userStore.token) {
    config.headers.Authorization = `Bearer ${userStore.token}`
  }
  return config
})

// 响应拦截器：统一错误处理
api.interceptors.response.use(
  response => {
    const { code, message, data } = response.data
    if (code !== 0) {
      // 业务错误
      if (code === 'E1004' || code === 'E1003') {
        // Token 过期或密码错误 → 清除登录态
        useUserStore().logout()
        router.push('/login')
      }
      return Promise.reject(new Error(message))
    }
    return data  // 直接返回 data，调用方可用 .data.xxx
  },
  error => {
    // 网络错误/超时
    if (error.code === 'ECONNABORTED') {
      toast.error('请求超时，请重试')
    } else if (!error.response) {
      toast.error('网络异常，请检查网络连接')
    } else {
      const status = error.response.status
      if (status === 401) { useUserStore().logout(); router.push('/login') }
      else if (status === 403) { toast.error('无权限') }
      else if (status === 500) { toast.error('服务器错误') }
    }
    return Promise.reject(error)
  }
)

export default api
```

### useRouteDriving 详细设计

**核心职责**:
1. 用双层 Polyline（outline + core）连接 waypoints，模拟高德原生路线风格
2. 创建起点/终点 Marker（起点=出发城市绿色标记，终点=最后一个 waypoint 红色标记）
3. 管理地图上的 Polyline + Marker 图层（逐方案增量添加）
4. 路线高亮/半透明切换（选中方案 vs 其他方案）
5. 出发城市 ≠ 第一个 waypoint 时画虚线连接

**城市坐标映射**: 内置 20 个潮汕周边城市坐标（深圳/广州/厦门/福州/汕头/潮州/揭阳/汕尾/珠海/东莞/惠州/佛山/中山/江门/肇庆/梅州/泉州/漳州/香港/澳门），未命中时调用 `AMap.Geocoder` 降级。

**方案颜色**: `PLAN_COLORS = { A: '#28C76F' (活力绿·美食), B: '#FF9F43' (琥珀橙·文化), C: '#0091FF' (科技蓝·综合) }`

**降级策略**: 因 AMap Key 无 Driving API 权限，完全移除 `AMap.Driving` 调用，用双层实线 Polyline + 方向箭头 (`showDir: true`) 直接连接 waypoints。
