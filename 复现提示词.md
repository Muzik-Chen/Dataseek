-- ============================================================
-- 兼容已有数据库的增量 DDL（用存储过程安全添加列）

DELIMITER //
CREATE PROCEDURE IF NOT EXISTS add_column_if_missing(
    IN tbl_name VARCHAR(128),
    IN col_name VARCHAR(128),
    IN col_def TEXT
)
BEGIN
    DECLARE col_count INT;
    SELECT COUNT(*) INTO col_count
    FROM information_schema.COLUMNS
    WHERE TABLE_SCHEMA = 'chaoshan_platform'
      AND TABLE_NAME = tbl_name
      AND COLUMN_NAME = col_name;
    IF col_count = 0 THEN
        SET @sql = CONCAT('ALTER TABLE ', tbl_name, ' ADD COLUMN ', col_name, ' ', col_def);
        PREPARE stmt FROM @sql;
        EXECUTE stmt;
        DEALLOCATE PREPARE stmt;
    END IF;
END //
DELIMITER ;

CALL add_column_if_missing('heritages', 'latitude',  'DECIMAL(10,7) DEFAULT NULL');

## 第四步：修复前端用户状态持久化

**文件：`chaoshangit/frontend/src/stores/user.js`**

完整替换为：

```javascript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUserStore = defineStore('user', () => {
  const token = ref(localStorage.getItem('token') || '')

  // 从 localStorage 恢复用户信息，避免刷新后丢失
  let savedUser = null
  try {
    const raw = localStorage.getItem('user')
    if (raw) savedUser = JSON.parse(raw)
  } catch { /* ignore parse error */ }
  const user = ref(savedUser)

  const isLoggedIn = computed(() => !!token.value)
  const isAdmin = computed(() => user.value?.role === 'admin')

  function setAuth(newToken, newUser) {
    token.value = newToken
    user.value = newUser
    localStorage.setItem('token', newToken)
    if (newUser) {
      localStorage.setItem('user', JSON.stringify(newUser))
    }
  }

  function logout() {
    token.value = ''
    user.value = null
    localStorage.removeItem('token')
    localStorage.removeItem('user')
  }

  return { token, user, isLoggedIn, isAdmin, setAuth, logout }
})
```

**id=1 潮州音乐（393字）：**
潮州音乐是潮汕地区传统音乐的统称，包括弦诗乐、笛套古乐、潮州锣鼓乐、庙堂音乐等多种形式，被誉为"唐宋遗音""华夏正声"，2006年入选第一批国家级非物质文化遗产名录。其历史可追溯至唐宋时期，中原古乐随历代移民南迁传入潮汕，与当地民间音乐融合，至今已有六七百年历史。潮州音乐以二弦为领奏乐器，辅以椰胡、扬琴、琵琶、三弦、古筝等，音色古朴典雅，旋律优美流畅，风格细腻精致，是中国传统音乐在民间的珍贵遗存。它采用独特的"二四谱"记谱法，有轻六调、重六调、活五调、反线调四大调式体系，演奏时讲究"加花""催奏""造句"等即兴技法，同一首乐曲在不同演奏者手中可呈现迥异的韵味。代表曲目"潮乐十大套"如《寒鸦戏水》《昭君怨》《平沙落雁》《小桃红》等，多为唐宋大曲和宋元套曲的遗响，是研究中国古代音乐的活化石。著名竹笛演奏家刘森曾感叹："要了解中国音乐，必须到潮州去，亲耳聆听那活着的古代音乐。"

英歌舞是潮汕地区独有的传统民间广场舞蹈，被誉为"中华战舞"，2006年入选第一批国家级非物质文化遗产名录。它以古老傩文化为精神内核，吸收北方秧歌的表演形式，融合南派武术和潮汕地方戏曲，以《水浒传》梁山好汉化妆卖艺攻打大名府营救卢俊义的故事为表现内容。表演者画着脸谱扮作一百零八好汉，身着英雄衣、双手各执一根英歌槌，在锣鼓声中边走边舞，动作刚猛豪迈、气势磅礴，挥槌齐眉、眼随槌动，跳如虎跃、转似旋龙。英歌舞分为慢板、中板、快板三种板式，慢板凝重古朴，中板稳健潇洒，快板迅猛凌厉，其中普宁快板英歌最具代表性。队形变化丰富多样，有"双龙出海""田螺圈""麦穗花""八字双金钱""打四门"等经典阵势。每逢春节和游神赛会，成百上千的英歌队员穿街过巷，寄托着扬正压邪、祈求国泰民安的朴素愿望，是潮汕最具群众基础和视觉冲击力的民俗艺术。

**id=3 潮剧（395字）：**
潮剧是用潮汕方言演唱的古老地方戏曲剧种，中国十大剧种之一、广东三大剧种之一，素有"南国奇葩"之美誉，2006年入选第一批国家级非物质文化遗产名录。它源于宋元南戏，是南戏在潮州方言区的地方化产物，距今已有近600年历史，流传着"有大海就有潮声，有潮人就有潮剧"的说法。潮剧唱腔优美、表演细腻，有轻六调、重六调、活五调、反线调等独特调式体系，用古谱"二四谱"记音，保留了较多唐宋古乐的遗韵。其丑行和彩罗衣旦最具特色，后台帮声形式为全国戏曲界所罕见——演员唱至精彩处，后台众声齐唱相和，气势动人。伴奏融合了弦诗、笛套、锣鼓等潮州音乐元素，管弦与打击配合和谐。经典剧目有《荔镜记》（陈三五娘）《苏六娘》《金花女》《扫窗会》等，传统剧目多达5000余个。潮剧不仅流行于粤东闽南，更随潮人足迹远播东南亚和欧美，梅兰芳、老舍、田汉等文化大家都曾为其倾倒，是联结海内外数千万潮汕人的情感纽带与文化认同符号。

**id=4 潮汕工夫茶（389字）：**
潮汕工夫茶是中国茶道的"活化石"，承袭唐宋瀹茶法，至清代中期已蔚然成风，2022年作为"中国传统制茶技艺及其相关习俗"的重要组成部分入选联合国教科文组织人类非物质文化遗产代表作名录。工夫茶以凤凰单丛为核心茶品，该茶产于潮州凤凰山海拔千米以上的古茶园，始于南宋末年，至今已有900多年历史，有"茶中香水"之美誉，香型涵盖桂花、茉莉、蜂蜜、芝兰等数十种，极为耐冲泡，一般可冲20余次而色香味不减。冲泡技艺极为考究，讲究"和、敬、精、乐"四字精神，核心程式包括备器、纳茶、高冲、刮沫淋壶、关公巡城、韩信点兵等步骤。茶具讲究"四宝"：玉书煨（烧水砂铫）、潮汕炉（红泥小炭炉）、孟臣罐（宜兴紫砂小壶）、若琛杯（景德镇小瓷杯）。在潮汕，茶叶被称为"茶米"，与大米相提并论，一句"有潮水的地方就有工夫茶"道出了它在潮人生活中无可替代的崇高地位。2023年，潮州获授"世界工夫茶文化之乡"称号。

**id=5 嵌瓷（372字）：**
嵌瓷是潮汕地区独有的传统建筑装饰技艺，与木雕、石雕并称潮汕古建筑"三绝"，被誉为"屋顶上的艺术"，2008年入选国家级非物质文化遗产名录。它始于明代万历年间，距今已有400多年历史。明代潮州陶瓷业发达，民间艺人发现废弃的彩瓷片可以用来装饰建筑，变废为宝，创造了这门独特工艺。清代随祠堂豪宅大量兴建而进入鼎盛期，并随潮人足迹传播至东南亚各地。嵌瓷以专门烧制的低温彩瓷碗为材料，经敲碎后由艺人用特制钳子剪取所需形状，再以贝灰泥（贝壳灰加红糖、草纸调制）粘贴于屋脊、檐下，创作出龙凤瑞兽、花卉鸟兽、戏曲人物等立体造型。潮汕有句俗语"厝角头有戏出"，正是指屋顶嵌瓷刻画着《三国》《水浒》等精彩戏剧故事。嵌瓷最大的特色是色彩斑斓且久经风雨烈日而不褪色，在亚热带高温多雨的气候条件下为其他装饰材料所无法替代，集中体现了潮汕工艺精致细腻、追求极致的美学精神。

**id=6 潮汕木雕（370字）：**
潮汕木雕又称潮州木雕、金漆木雕，与浙江东阳木雕、乐清黄杨木雕、福建龙眼木雕并誉为"中国四大木雕"，2006年入选第一批国家级非物质文化遗产名录。其历史可追溯至唐代，潮州开元寺保存的唐代木鱼为现存最早实物，至今已有千余年传承。潮汕木雕以樟木为主要材料，质地柔韧细腻且带樟脑香气能防虫蛀。其最大的特色是髹漆贴金——雕刻完成后涂以植物漆调银朱色粉为"朱金地漆"，再贴上24K金箔，作品金碧辉煌、数百年不褪色且愈显金黄。技法以多层镂通雕最为精湛，可达二至十层，能将人物众多、情节复杂的故事浓缩于方寸之间，展现亭台楼阁、山峦曲径的空间纵深，达到"细而不腻、大而不空、繁而不乱、艳而不俗"的美学境界。题材涵盖潮剧戏出、历史典故、水族海鲜（龙虾蟹篓最具代表性）、花鸟瑞兽等，广泛应用于祠堂庙宇、神器礼器和家具装饰，承载着潮汕民间信仰与宗族文化的深厚底蕴。

**id=7 铁枝木偶（364字）：**
铁枝木偶俗称"纸影戏"，由南宋时期随中原移民传入潮汕的皮影戏演变而来，距今约有700年历史，2006年入选第一批国家级非物质文化遗产名录。它经历了从平面皮影到立体木偶的创造性转化：清末艺人改影现为形现，将牛皮雕形改为木偶造型，"捆草为身、扎纸为手、削木为足、塑泥为头"，傀儡从平面走向立体。因采用三根铁枝安在木偶背后和双手进行操控，故定名为"铁枝木偶"。表演时后台三至四人盘膝而坐，以推、拉、提、拨、抖等手法操纵，可完成飞天、入地、点火、射箭、骑马、武打等复杂动作，其剧目、唱腔与潮剧完全相同，故有"微型潮剧"之称。一台铁枝木偶戏融大吴泥塑（偶头）、潮绣（戏服）、潮剧（唱腔）等多项非遗于一身，潮汕谣谚"手打鼓、脚打锣、口唱歌、头壳撞深波"生动概括了它一人多角、手脚并用的表演特色，是研究中国戏曲演变史不可多得的珍贵活化石。

**id=8 潮汕剪纸（374字）：**
潮汕剪纸是广东剪纸的重要流派，包括潮阳剪纸和潮州剪纸两大分支，2006年以"广东剪纸"名义列入第一批国家级非物质文化遗产名录。其历史可追溯至东晋时期，由南迁的中原移民传入潮汕，至今已有约1500年传承。潮汕剪纸以"花中套花"的构图布局为核心特色，结构严谨饱满而不杂乱，玲珑剔透、疏密有致。其最大的技艺亮点是"潜剪"技法——剪刀从纸背运剪，在纸面下畅快游走，线条流畅如丝而遒劲奔放，细如发丝而不显柔弱。传统艺人无需描摹纹样，心中构思即可直接开剪，拙朴自然，随意性强。作品多用于游神赛会、元宵观灯、中秋拜月、婚俗喜庆、祭祖拜神等民俗活动，"图必有意，意必吉祥"，以谐音和象征手法寄托着五谷丰登、子嗣绵延、福寿安康等美好愿望。代表性传承人许遵英（许舜英）12岁学艺，擅长"潜剪"和"随手入形"，作品多次获国家级金奖，被中国美术馆、广东省美术馆等机构收藏。

**id=9 潮州大锣鼓（374字）：**
潮州大锣鼓是潮汕地区最具代表性的民间打击乐合奏形式，被誉为"东方交响乐"，2006年作为潮州音乐的重要组成部分入选第一批国家级非物质文化遗产名录。其历史可追溯至唐代，韩愈《潮州祭神文》中已有"吹击管鼓"的记载，明清时期随戏曲兴盛而形成固定表演模式。潮州大锣鼓以司鼓为中心，大鼓有30多种不同击法和40多种指挥手势，辅以苏锣、深波、斗锣、大钹等十余种打击乐器，以及唢呐、椰胡、二胡等管弦乐器，全队编制约50人。分为行进演奏的"长行套"（用于游神赛会游行，穿街过巷、声震百里）和坐定演奏的"牌子套"（用于舞台演出），传统曲目有十八大套如《抛网捕鱼》《陈春生告官》等。1957年在莫斯科第六届世界青年联欢节上获金质奖章，从此蜚声中外。在潮汕几乎每个自然村都有自己的大锣鼓班，"游神赛会没有锣鼓队，有如宴席没有酒"这句俗语道出了它在潮人心中不可替代的地位。

**id=10 普宁英歌（350字）：**
普宁英歌是英歌舞中快板流派的杰出代表，与潮阳英歌同源异流而风格更为粗犷迅猛，2006年作为英歌舞的重要组成部分入选第一批国家级非物质文化遗产名录。普宁英歌以快节奏著称，英歌槌仅长约35至40公分，比潮阳英歌的槌更短更轻，动作灵活迅猛、快击快收，节奏紧凑激烈，气势如虹，在三种板式中最具视觉冲击力。代表性传承人陈来发率领的普宁南山英歌队平均年龄仅约25岁，将这门古老艺术注入了青春活力，打破了"传内不传外、传男不传女"的传统，积极推动进校园传承，并涌现出多支女子英歌队。近年来普宁英歌多次受邀赴英国伦敦、泰国曼谷、南非开普敦等地演出，从地方民俗一跃成为中华文化对外交流的亮丽名片。每到春节，仅普宁就有上百支英歌队走街串巷，以刚猛豪迈的舞姿和震天的锣鼓声点燃潮汕大地的新春狂欢，被誉为"过年最强气氛组"。
